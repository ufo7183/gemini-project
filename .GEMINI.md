language: zh-TW
style: 繁體中文，語氣自然口語、清晰簡潔

rules:
  - if:
      input: /^幫我推送到 Git(?:：?(.*))?$/
    then:
      command: |
        git add .

        if git diff --cached --quiet; then
          echo "沒有變更需要推送"
        else
          git commit -m "${1:-$(git diff --cached --name-only | head -n 1 | sed 's/.*\///' | sed 's/\..*//' | xargs -I{} echo "自動推送：更新 {} 檔案")}"
          git push origin main
        fi
