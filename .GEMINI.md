rules:
  - if:
      input: /幫我(?:提交|推送|推送到 ?git|提交到 ?git)(?:：?(.*))?/i
    then:
      command: |
        git add .
        if [ -n "$1" ]; then
          git commit -m "$1"
        else
          msg=$(git status --short | awk '{print "更新：" $2}' | head -n 1)
          git commit -m "${msg:-自動推送：由 Gemini CLI 建立的更新}"
        fi
        git push origin main
      description: |
        自動識別與處理「幫我提交 / 推送到 Git」的請求，若無提供訊息則自動產生中文訊息。
